# 1) Install system Chromium (Ubuntu/Debian)
sudo apt update
sudo apt install -y chromium
which chromium     # expect /usr/bin/chromium (not /snap/bin/chromium)
ls -la /usr/bin/chromium

# 2) If only /snap/bin/chromium is found, install the .deb explicitly:
#    sudo apt install -y chromium-browser   # creates /usr/bin/chromium-browser
#    or grab Google Chrome stable:
#    wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
#    sudo apt install -y ./google-chrome-stable_current_amd64.deb
#    which google-chrome-stable

# 3) (Optional) Pin the binary in /home/sasha/apps/cheradip/bcheradip/.env
echo "PLAYWRIGHT_CHROMIUM_EXECUTABLE_PATH=/usr/bin/chromium" | sudo tee -a /home/sasha/apps/cheradip/bcheradip/.env

# 4) Pull the new backend code, then restart
cd /home/sasha/apps/cheradip/bcheradip
git fetch origin
sudo systemctl restart cheradip

sudo apt install -y libnss3 libnspr4 libdrm2 libgbm1 libxkbcommon0 \
    libasound2t64 libatk1.0-0t64 libatk-bridge2.0-0t64 libcups2t64 \
    libatspi2.0-0t64 libxcomposite1 libxdamage1 libxfixes3 libxrandr2 \
    libpango-1.0-0 libcairo2

cd /home/sasha/apps/cheradip/bcheradip
git fetch origin
git status                     # see what's "ahead" / "behind"
git branch --show-current      # confirm branch name (likely "main")
git reset --hard origin/main   # discard server-side local commits, take origin
git log -1 --oneline           # verify you now have the latest commit

sudo systemctl restart cheradip
sudo systemctl status cheradip --no-pager
